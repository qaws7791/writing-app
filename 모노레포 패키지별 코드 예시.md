# 모노레포 패키지별 코드 예시

## 의존성 흐름

`apps/web` → `apps/api` → `packages/core` ← `packages/db`, `packages/storage`

핵심 원칙: **core는 프레임워크/DB/네트워크에 독립적이되, 검증된 범용 FP 유틸리티(`neverthrow`, `remeda`, `ts-pattern`)에는 직접 의존한다.** db와 storage는 core가 정의한 포트(함수 타입)를 구현한다. api는 모든 것을 조합(compose)한다.

---

## 1. packages/core

비즈니스 로직의 심장. 런타임 프레임워크에는 의존하지 않지만, 안정적인 범용 FP 유틸 라이브러리는 직접 사용한다.

### `Result 규약` — `neverthrow`를 직접 사용

```typescript
import { err, ok, ResultAsync, type Result } from "neverthrow";

// Result/ResultAsync를 자체 래핑하지 않는다.
// 모든 패키지는 neverthrow를 직접 import해서 같은 표준 API를 사용한다.

const parseJson = (raw: string): Result<unknown, string> => {
  try {
    return ok(JSON.parse(raw));
  } catch (error) {
    return err(error instanceof Error ? error.message : "JSON 파싱 실패");
  }
};

const readConfig = (path: string) =>
  ResultAsync.fromPromise(
    Bun.file(path).text(),
    () => `설정 파일을 읽을 수 없습니다: ${path}`,
  ).andThen(parseJson);
```

### `src/types/brand.ts` — 브랜드 타입으로 원시 타입에 의미 부여

```typescript
// "Parse, don't validate" (Alexis King / Code That Fits in Your Head)
// 원시 타입을 도메인 의미로 래핑하여 컴파일 타임에 혼동을 방지

declare const __brand: unique symbol;

export type Brand<T, B extends string> = T & {
  readonly [__brand]: B;
};

export type UserId = Brand<string, "UserId">;
export type ArticleId = Brand<string, "ArticleId">;
export type Email = Brand<string, "Email">;
export type Slug = Brand<string, "Slug">;
export type FileKey = Brand<string, "FileKey">;

// 브랜드 생성 헬퍼 — 검증 후에만 호출해야 한다
export const unsafeBrand = <B extends Brand<any, any>>(
  value: B extends Brand<infer T, any> ? T : never,
): B => value as B;
```

### `src/validators/article.ts` — 입력 검증 (순수 함수)

```typescript
import type { CreateArticleInput, UpdateArticleInput } from "../types/article";
import type { Slug } from "../types/brand";
import type { ValidationError } from "../types/errors";
import { err, ok, Result, type Result as ResultType } from "neverthrow";
import { map, pipe, unique } from "remeda";
import { validationError } from "../types/errors";
import { unsafeBrand } from "../types/brand";

// 검증은 파싱이다 — string을 받아 도메인 타입을 반환한다
// 각 검증 함수는 작고, 독립적이며, 조합 가능하다

const TITLE_MIN_LENGTH = 3;
const TITLE_MAX_LENGTH = 200;
const CONTENT_MIN_LENGTH = 10;
const CONTENT_MAX_LENGTH = 50_000;
const MAX_TAGS = 10;
const TAG_MAX_LENGTH = 30;
const SLUG_PATTERN = /^[a-z0-9]+(?:-[a-z0-9]+)*$/;

export const validateTitle = (
  title: string,
): ResultType<string, ValidationError> => {
  const trimmed = title.trim();
  if (trimmed.length < TITLE_MIN_LENGTH) {
    return err(
      validationError(
        "title",
        `제목은 최소 ${TITLE_MIN_LENGTH}자 이상이어야 합니다`,
      ),
    );
  }
  if (trimmed.length > TITLE_MAX_LENGTH) {
    return err(
      validationError(
        "title",
        `제목은 최대 ${TITLE_MAX_LENGTH}자까지 가능합니다`,
      ),
    );
  }
  return ok(trimmed);
};

export const validateContent = (
  content: string,
): ResultType<string, ValidationError> => {
  if (content.length < CONTENT_MIN_LENGTH) {
    return err(
      validationError(
        "content",
        `본문은 최소 ${CONTENT_MIN_LENGTH}자 이상이어야 합니다`,
      ),
    );
  }
  if (content.length > CONTENT_MAX_LENGTH) {
    return err(
      validationError(
        "content",
        `본문은 최대 ${CONTENT_MAX_LENGTH}자까지 가능합니다`,
      ),
    );
  }
  return ok(content);
};

export const validateTags = (
  tags: readonly string[],
): ResultType<readonly string[], ValidationError> => {
  if (tags.length > MAX_TAGS) {
    return err(
      validationError("tags", `태그는 최대 ${MAX_TAGS}개까지 가능합니다`),
    );
  }
  const normalized = pipe(
    tags,
    map((tag) => tag.trim().toLowerCase()),
    unique(),
  );
  const invalid = normalized.find((tag) => tag.length > TAG_MAX_LENGTH);
  if (invalid) {
    return err(
      validationError(
        "tags",
        `태그 "${invalid}"이(가) 최대 길이를 초과합니다`,
      ),
    );
  }
  return ok(normalized);
};

// 제목에서 슬러그 생성 — 순수 변환 함수
export const toSlug = (title: string): Slug => {
  const slug = title
    .toLowerCase()
    .replace(/[^\w\s가-힣-]/g, "")
    .replace(/[\s_]+/g, "-")
    .replace(/^-+|-+$/g, "")
    .slice(0, 100);

  const timestampSuffix = Date.now().toString(36);
  return unsafeBrand<Slug>(`${slug}-${timestampSuffix}`);
};

export const validateSlug = (
  slug: string,
): ResultType<Slug, ValidationError> => {
  if (!SLUG_PATTERN.test(slug)) {
    return err(
      validationError("slug", "슬러그는 소문자, 숫자, 하이픈만 가능합니다"),
    );
  }
  return ok(unsafeBrand<Slug>(slug));
};

// 전체 생성 입력 검증 — 작은 검증 함수들의 조합
export type ValidatedCreateArticleInput = {
  readonly title: string;
  readonly slug: Slug;
  readonly content: string;
  readonly tags: readonly string[];
  readonly coverImageKey: string | null;
};

export const validateCreateArticleInput = (
  input: CreateArticleInput,
): ResultType<ValidatedCreateArticleInput, ValidationError> =>
  Result.combine([
    validateTitle(input.title),
    validateContent(input.content),
    validateTags(input.tags),
  ] as const).map(([title, content, tags]) => ({
    title,
    slug: toSlug(title),
    content,
    tags,
    coverImageKey: input.coverImageKey ?? null,
  }));
```

### `src/validators/user.ts`

```typescript
import type { CreateUserInput, UserRole } from "../types/user";
import type { Email } from "../types/brand";
import type { ValidationError } from "../types/errors";
import { err, ok, Result, type Result as ResultType } from "neverthrow";
import { validationError } from "../types/errors";
import { unsafeBrand } from "../types/brand";

const EMAIL_PATTERN = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
const NAME_MIN_LENGTH = 2;
const NAME_MAX_LENGTH = 100;
const VALID_ROLES: readonly UserRole[] = ["admin", "author", "reader"];

export const validateEmail = (
  email: string,
): ResultType<Email, ValidationError> => {
  const trimmed = email.trim().toLowerCase();
  if (!EMAIL_PATTERN.test(trimmed)) {
    return err(
      validationError("email", "올바른 이메일 형식이 아닙니다"),
    );
  }
  return ok(unsafeBrand<Email>(trimmed));
};

export const validateName = (
  name: string,
): ResultType<string, ValidationError> => {
  const trimmed = name.trim();
  if (trimmed.length < NAME_MIN_LENGTH || trimmed.length > NAME_MAX_LENGTH) {
    return err(
      validationError(
        "name",
        `이름은 ${NAME_MIN_LENGTH}~${NAME_MAX_LENGTH}자여야 합니다`,
      ),
    );
  }
  return ok(trimmed);
};

export const validateRole = (
  role: string,
): ResultType<UserRole, ValidationError> => {
  if (!VALID_ROLES.includes(role as UserRole)) {
    return err(validationError("role", `유효하지 않은 역할: ${role}`));
  }
  return ok(role as UserRole);
};

export type ValidatedCreateUserInput = {
  readonly email: Email;
  readonly name: string;
  readonly role: UserRole;
};

export const validateCreateUserInput = (
  input: CreateUserInput,
): ResultType<ValidatedCreateUserInput, ValidationError> =>
  Result.combine([
    validateEmail(input.email),
    validateName(input.name),
    validateRole(input.role),
  ] as const).map(([email, name, role]) => ({
    email,
    name,
    role,
  }));
```

### `src/rules/article.ts` — 비즈니스 규칙 (순수 술어 함수)

```typescript
import type {
  Article,
  DraftArticle,
  PublishedArticle,
  ArchivedArticle,
} from "../types/article";
import type { User } from "../types/user";
import type { BusinessRuleError } from "../types/errors";
import { err, ok, type Result } from "neverthrow";
import { businessRuleError } from "../types/errors";

// 비즈니스 규칙은 순수 함수(술어 함수)로 표현한다
// 각 규칙은 하나의 관심사만 다룬다 (SRP)
// 규칙의 이름이 곧 문서다 (The Art of Readable Code)

export const isDraft = (article: Article): article is DraftArticle =>
  article.status === "draft";

export const isPublished = (
  article: Article,
): article is PublishedArticle => article.status === "published";

export const isArchived = (
  article: Article,
): article is ArchivedArticle => article.status === "archived";

export const isAuthor = (article: Article, user: User): boolean =>
  article.authorId === user.id;

export const canEdit = (article: Article, user: User): boolean =>
  isDraft(article) && (isAuthor(article, user) || user.role === "admin");

export const canPublish = (article: Article, user: User): boolean =>
  isDraft(article) && (isAuthor(article, user) || user.role === "admin");

export const canArchive = (article: Article, user: User): boolean =>
  isPublished(article) &&
  (isAuthor(article, user) || user.role === "admin");

// 상태 전이 함수 — 불변 데이터를 새 데이터로 변환
// 전이가 불가능하면 Result로 에러를 반환

export const publish = (
  article: Article,
  user: User,
  now: Date,
): Result<PublishedArticle, BusinessRuleError> => {
  if (!isDraft(article)) {
    return err(
      businessRuleError(
        "publish",
        `초안 상태의 기사만 출판할 수 있습니다 (현재: ${article.status})`,
      ),
    );
  }
  if (!canPublish(article, user)) {
    return err(
      businessRuleError("publish", "이 기사를 출판할 권한이 없습니다"),
    );
  }
  if (article.content.length < 50) {
    return err(
      businessRuleError(
        "publish",
        "출판하려면 본문이 50자 이상이어야 합니다",
      ),
    );
  }

  return ok({
    ...article,
    status: "published" as const,
    publishedAt: now,
    updatedAt: now,
  });
};

export const archive = (
  article: Article,
  user: User,
  now: Date,
): Result<ArchivedArticle, BusinessRuleError> => {
  if (!isPublished(article)) {
    return err(
      businessRuleError(
        "archive",
        `출판된 기사만 보관할 수 있습니다 (현재: ${article.status})`,
      ),
    );
  }
  if (!canArchive(article, user)) {
    return err(
      businessRuleError("archive", "이 기사를 보관할 권한이 없습니다"),
    );
  }

  return ok({
    ...article,
    status: "archived" as const,
    archivedAt: now,
    updatedAt: now,
  });
};

// 초안 수정 — 초안 상태에서만 가능
export const updateDraft = (
  article: DraftArticle,
  changes: {
    readonly title?: string;
    readonly content?: string;
    readonly tags?: readonly string[];
    readonly coverImageKey?: string | null;
  },
  now: Date,
): DraftArticle => ({
  ...article,
  ...(changes.title !== undefined && { title: changes.title }),
  ...(changes.content !== undefined && { content: changes.content }),
  ...(changes.tags !== undefined && { tags: changes.tags }),
  ...(changes.coverImageKey !== undefined && {
    coverImageKey: changes.coverImageKey,
  }),
  updatedAt: now,
});
```

```typescript
// 현재 시각 의존성을 명시적으로 만들어 테스트 가능하게 한다
// (Growing Object-Oriented Software Guided by Tests)

export const systemClock: Clock = {
```

```typescript
import type { InferSelectModel } from "drizzle-orm";
import type { Article, DraftArticle, PublishedArticle, ArchivedArticle } from "@repo/core";
import { err, ok, type Result } from "neverthrow";
import { match } from "ts-pattern";
import { unsafeBrand, repositoryError } from "@repo/core";
import type { DomainError } from "@repo/core";
import type { FileKey, ArticleId, UserId, Slug } from "@repo/core";
import { articles } from "../schema/articles";

// DB에서 조회된 행의 타입
type ArticleRow = InferSelectModel<typeof articles>;

// 매퍼는 순수 함수다 — DB 표현과 도메인 표현 사이를 변환한다
// 데이터 스키마와 도메인 모델을 분리한다 (DOP 원칙 4)

export const toDomain = (row: ArticleRow): Result<Article, DomainError> => {
  const base = {
    id: unsafeBrand<ArticleId>(row.id),
    title: row.title,
    slug: unsafeBrand<Slug>(row.slug),
    content: row.content,
    authorId: unsafeBrand<UserId>(row.authorId),
    coverImageKey: row.coverImageKey
      ? unsafeBrand<FileKey>(row.coverImageKey)
      : null,
    tags: parseTags(row.tags),
    createdAt: row.createdAt,
    updatedAt: row.updatedAt,
  };

  return match(row)
    .with({ status: "draft" }, () =>
      ok({ ...base, status: "draft" as const } satisfies DraftArticle),
    )
    .with({ status: "published", publishedAt: null }, () =>
      err(repositoryError("map", "출판된 기사에 publishedAt이 없습니다")),
    )
    .with({ status: "published" }, (publishedRow) =>
      ok({
        ...base,
        status: "published" as const,
        publishedAt: publishedRow.publishedAt,
      } satisfies PublishedArticle),
    )
    .with({ status: "archived", publishedAt: null }, () =>
      err(
        repositoryError(
          "map",
          "보관된 기사에 publishedAt 또는 archivedAt이 없습니다",
        ),
      ),
    )
    .with({ status: "archived", archivedAt: null }, () =>
      err(
        repositoryError(
          "map",
          "보관된 기사에 publishedAt 또는 archivedAt이 없습니다",
        ),
      ),
    )
    .with({ status: "archived" }, (archivedRow) =>
      ok({
        ...base,
        status: "archived" as const,
        publishedAt: archivedRow.publishedAt,
        archivedAt: archivedRow.archivedAt,
      } satisfies ArchivedArticle),
    )
    .exhaustive();
};

// 도메인 → DB 행 변환
export const toRow = (
  article: Article,
): Omit<ArticleRow, never> => ({
  id: article.id,
  title: article.title,
  slug: article.slug,
  content: article.content,
  status: article.status,
  authorId: article.authorId,
  coverImageKey: article.coverImageKey,
  tags: JSON.stringify(article.tags),
  createdAt: article.createdAt,
  updatedAt: article.updatedAt,
  publishedAt:
    article.status === "published" || article.status === "archived"
      ? article.publishedAt
      : null,
  archivedAt:
    article.status === "archived" ? article.archivedAt : null,
});

const parseTags = (tagsJson: string): readonly string[] => {
  try {
    const parsed = JSON.parse(tagsJson);
    return Array.isArray(parsed) ? parsed : [];
  } catch {
    return [];
  }
};
```

```typescript
import { eq, and, desc, sql } from "drizzle-orm";
import { err, ok, Result } from "neverthrow";
import { map, pipe } from "remeda";
import type {
  ArticleRepository,
  Article,
  ArticleFilter,
  ArticleSummary,
} from "@repo/core";
import {
  notFoundError,
  repositoryError,
  unsafeBrand,
} from "@repo/core";
import type { ArticleId, Slug } from "@repo/core";
import type { DrizzleDB } from "../connection";
import { articles } from "../schema/articles";
import { users } from "../schema/users";
import { articleMapper } from "../mappers";

// 포트 구현은 일반 객체 팩토리 함수다
// class 없이, new 없이, this 없이 — DOP/FP 스타일

export const createArticleRepository = (
  db: DrizzleDB,
): ArticleRepository => ({
  findById: async (id) => {
    try {
      const row = await db
        .select()
        .from(articles)
        .where(eq(articles.id, id))
        .get();

      if (!row) return err(notFoundError("Article", id));
      return articleMapper.toDomain(row);
    } catch (error) {
      return err(
        repositoryError("findById", errorMessage(error)),
      );
    }
  },

  findBySlug: async (slug) => {
    try {
      const row = await db
        .select()
        .from(articles)
        .where(eq(articles.slug, slug))
        .get();

      if (!row) return err(notFoundError("Article", slug));
      return articleMapper.toDomain(row);
    } catch (error) {
      return err(
        repositoryError("findBySlug", errorMessage(error)),
      );
    }
  },

  findMany: async (filter) => {
    try {
      const conditions = buildConditions(filter);
      const rows = await db
        .select()
        .from(articles)
        .where(conditions.length > 0 ? and(...conditions) : undefined)
        .orderBy(desc(articles.createdAt))
        .limit(filter.limit ?? 20)
        .offset(filter.offset ?? 0)
        .all();

      return Result.combine(pipe(rows, map(articleMapper.toDomain))).mapErr(() =>
        repositoryError("findMany", "일부 행의 매핑에 실패했습니다"),
      );
    } catch (error) {
      return err(
        repositoryError("findMany", errorMessage(error)),
      );
    }
  },

  findSummaries: async (filter) => {
    try {
      const conditions = buildConditions(filter);
      const rows = await db
        .select({
          id: articles.id,
          title: articles.title,
          slug: articles.slug,
          status: articles.status,
          authorName: users.name,
          tags: articles.tags,
          createdAt: articles.createdAt,
          publishedAt: articles.publishedAt,
        })
        .from(articles)
        .innerJoin(users, eq(articles.authorId, users.id))
        .where(conditions.length > 0 ? and(...conditions) : undefined)
        .orderBy(desc(articles.createdAt))
        .limit(filter.limit ?? 20)
        .offset(filter.offset ?? 0)
        .all();

      const summaries: ArticleSummary[] = pipe(
        rows,
        map((row) => ({
          id: unsafeBrand(row.id),
          title: row.title,
          slug: unsafeBrand(row.slug),
          status: row.status as Article["status"],
          authorName: row.authorName,
          tags: JSON.parse(row.tags),
          createdAt: row.createdAt,
          publishedAt: row.publishedAt,
        })),
      );

      return ok(summaries);
    } catch (error) {
      return err(
        repositoryError("findSummaries", errorMessage(error)),
      );
    }
  },

  save: async (article) => {
    try {
      const row = articleMapper.toRow(article);
      await db
        .insert(articles)
        .values(row)
        .onConflictDoUpdate({
          target: articles.id,
          set: {
            title: row.title,
            slug: row.slug,
            content: row.content,
            status: row.status,
            coverImageKey: row.coverImageKey,
            tags: row.tags,
            updatedAt: row.updatedAt,
            publishedAt: row.publishedAt,
            archivedAt: row.archivedAt,
          },
        });

      return ok(article);
    } catch (error) {
      return err(
        repositoryError("save", errorMessage(error)),
      );
    }
  },

  remove: async (id) => {
    try {
      await db.delete(articles).where(eq(articles.id, id));
      return ok(undefined);
    } catch (error) {
      return err(
        repositoryError("remove", errorMessage(error)),
      );
    }
  },

  existsBySlug: async (slug) => {
    try {
      const row = await db
        .select({ id: articles.id })
        .from(articles)
        .where(eq(articles.slug, slug))
        .get();

      return ok(row !== undefined);
    } catch (error) {
      return err(
        repositoryError("existsBySlug", errorMessage(error)),
      );
    }
  },
});

// 필터에서 Drizzle 조건을 생성하는 헬퍼
const buildConditions = (filter: ArticleFilter) =>
  [
    filter.status ? eq(articles.status, filter.status) : undefined,
    filter.authorId ? eq(articles.authorId, filter.authorId) : undefined,
    filter.tag ? sql`json_each(${articles.tags}) = ${filter.tag}` : undefined,
  ].filter((condition): condition is NonNullable<typeof condition> => condition !== undefined);

const errorMessage = (error: unknown): string =>
  error instanceof Error ? error.message : String(error);
```

```typescript
      const query = db.select().from(users);
      const rows = filter.role
        ? await query.where(eq(users.role, filter.role)).all()
        : await query.all();

      return ok(rows.map(userMapper.toDomain));
    } catch (error) {
      return err(repositoryError("findMany", String(error)));
    }
  },

  save: async (user) => {
    try {
      const row = userMapper.toRow(user);
      await db
        .insert(users)
        .values(row)
        .onConflictDoUpdate({
          target: users.id,
          set: { name: row.name, role: row.role },
        });

      return ok(user);
    } catch (error) {
      return err(repositoryError("save", String(error)));
    }
  },

  remove: async (id) => {
    try {
      await db.delete(users).where(eq(users.id, id));
      return ok(undefined);
    } catch (error) {
      return err(repositoryError("remove", String(error)));
    }
  },

  existsByEmail: async (email) => {
    try {
      const row = await db
        .select({ id: users.id })
        .from(users)
        .where(eq(users.email, email))
        .get();

      return ok(row !== undefined);
    } catch (error) {
      return err(repositoryError("existsByEmail", String(error)));
    }
  },
});
```

```typescript
import {
  PutObjectCommand,
  GetObjectCommand,
  DeleteObjectCommand,
  HeadObjectCommand,
} from "@aws-sdk/client-s3";
import { getSignedUrl } from "@aws-sdk/s3-request-presigner";
import type { S3Client } from "@aws-sdk/client-s3";
import { err, ok } from "neverthrow";
import type { StoragePort, FileMetadata } from "@repo/core";
import type { FileKey } from "@repo/core";
import { storageError, unsafeBrand } from "@repo/core";

// 포트 구현은 팩토리 함수로 생성하는 일반 객체다
// class 없음 — 의존성은 클로저로 캡처

export const createStorageAdapter = (
  client: S3Client,
  bucket: string,
): StoragePort => ({
  upload: async (params) => {
    try {
      const command = new PutObjectCommand({
        Bucket: bucket,
        Key: params.key,
        Body: params.body as any,
        ContentType: params.contentType,
        Metadata: params.metadata,
      });

      await client.send(command);
      return ok(unsafeBrand<FileKey>(params.key));
    } catch (error) {
      return err(
        storageError("upload", errorMessage(error)),
      );
    }
  },

  download: async (key) => {
    try {
      const command = new GetObjectCommand({
        Bucket: bucket,
        Key: key,
      });

      const response = await client.send(command);

      if (!response.Body) {
        return err(storageError("download", "응답 본문이 비어있습니다"));
      }

      return ok({
        body: response.Body.transformToWebStream(),
        contentType: response.ContentType ?? "application/octet-stream",
      });
    } catch (error) {
      return err(
        storageError("download", errorMessage(error)),
      );
    }
  },

  remove: async (key) => {
    try {
      const command = new DeleteObjectCommand({
        Bucket: bucket,
        Key: key,
      });

      await client.send(command);
      return ok(undefined);
    } catch (error) {
      return err(
        storageError("remove", errorMessage(error)),
      );
    }
  },

  getSignedUrl: async (key, expiresInSeconds) => {
    try {
      const command = new GetObjectCommand({
        Bucket: bucket,
        Key: key,
      });

      const url = await getSignedUrl(client, command, {
        expiresIn: expiresInSeconds,
      });

      return ok(url);
    } catch (error) {
      return err(
        storageError("getSignedUrl", errorMessage(error)),
      );
    }
  },

  getMetadata: async (key) => {
    try {
      const command = new HeadObjectCommand({
        Bucket: bucket,
        Key: key,
      });

      const response = await client.send(command);

      return ok({
        key,
        contentType: response.ContentType ?? "application/octet-stream",
        size: response.ContentLength ?? 0,
        lastModified: response.LastModified ?? new Date(),
      });
    } catch (error) {
      return err(
        storageError("getMetadata", errorMessage(error)),
      );
    }
  },

  exists: async (key) => {
    try {
      const command = new HeadObjectCommand({
        Bucket: bucket,
        Key: key,
      });

      await client.send(command);
      return ok(true);
    } catch (error: any) {
      if (error?.name === "NotFound" || error?.$metadata?.httpStatusCode === 404) {
        return ok(false);
      }
      return err(
        storageError("exists", errorMessage(error)),
      );
    }
  },
});

const errorMessage = (error: unknown): string =>
  error instanceof Error ? error.message : String(error);
```

### `src/in-memory-adapter.ts` — 테스트용 인메모리 구현

```typescript
import type { StoragePort } from "@repo/core";
import { err, ok } from "neverthrow";
import type { FileKey } from "@repo/core";
import { storageError, unsafeBrand } from "@repo/core";

// 테스트에서 S3를 사용하지 않기 위한 인메모리 구현
// 포트와 어댑터 패턴의 장점: 구현을 자유롭게 교체 가능

type StoredFile = {
  readonly body: Uint8Array;
  readonly contentType: string;
  readonly metadata: Record<string, string>;
  readonly storedAt: Date;
};

export const createInMemoryStorage = (): StoragePort & {
  readonly getStore: () => ReadonlyMap<string, StoredFile>;
  readonly clear: () => void;
} => {
  const store = new Map<string, StoredFile>();

  return {
    upload: async (params) => {
      const body =
        params.body instanceof Uint8Array
          ? params.body
          : params.body instanceof Buffer
            ? new Uint8Array(params.body)
            : new Uint8Array(
                await new Response(params.body as ReadableStream).arrayBuffer(),
              );

      store.set(params.key, {
        body,
        contentType: params.contentType,
        metadata: params.metadata ?? {},
        storedAt: new Date(),
      });

      return ok(unsafeBrand<FileKey>(params.key));
    },

    download: async (key) => {
      const file = store.get(key);
      if (!file) {
        return err(storageError("download", `파일을 찾을 수 없습니다: ${key}`));
      }

      const stream = new ReadableStream({
        start(controller) {
          controller.enqueue(file.body);
          controller.close();
        },
      });

      return ok({ body: stream, contentType: file.contentType });
    },

    remove: async (key) => {
      store.delete(key);
      return ok(undefined);
    },

    getSignedUrl: async (key, _expiresInSeconds) => {
      if (!store.has(key)) {
        return err(
          storageError("getSignedUrl", `파일을 찾을 수 없습니다: ${key}`),
        );
      }
      return ok(`http://localhost/signed/${key}`);
    },

    getMetadata: async (key) => {
      const file = store.get(key);
      if (!file) {
        return err(
          storageError("getMetadata", `파일을 찾을 수 없습니다: ${key}`),
        );
      }
      return ok({
        key: unsafeBrand<FileKey>(key),
        contentType: file.contentType,
        size: file.body.length,
        lastModified: file.storedAt,
      });
    },

    exists: async (key) => ok(store.has(key)),

    getStore: () => store,
    clear: () => store.clear(),
  };
};
```

```typescript
) =>

    (error) => c.json(error, toHttpStatus(error) as any),
```

```typescript
import type { Context, Next } from "hono";

// 간소화된 인증 미들웨어
// 실제로는 JWT 검증 등을 수행

export type AuthUser = {
  readonly id: string;
  readonly role: string;
};

export const authMiddleware = (secretKey: string) => {
  return async (c: Context, next: Next) => {
    const authHeader = c.req.header("Authorization");

    if (!authHeader?.startsWith("Bearer ")) {
      return c.json({ code: "AUTH_ERROR", message: "인증이 필요합니다" }, 401);
    }

    const token = authHeader.slice(7);

    try {
      // 실제 환경에서는 JWT 검증
      // 여기서는 간소화를 위해 base64 디코딩
      const payload = JSON.parse(
        Buffer.from(token, "base64").toString(),
      ) as AuthUser;

      c.set("user", payload);
      await next();
    } catch {
      return c.json(
        { code: "AUTH_ERROR", message: "유효하지 않은 토큰입니다" },
        401,
      );
    }
  };
};
```

```typescript
const request = async <T>(
  path: string,
  options: RequestInit = {},
): Promise<ApiResponse<T>> => {

      ...options,
      headers: {
        "Content-Type": "application/json",
        ...options.headers,
      },
    });

    const body = await response.json();

    if (!response.ok) {
      return { ok: false, error: body as DomainError };
    }

    return { ok: true, data: body as T };
  } catch (error) {
    return {
      ok: false,
      error: {
        code: "REPOSITORY_ERROR" as const,
        operation: "fetch",
        message: error instanceof Error ? error.message : "네트워크 오류",
      },
    };

};
```

```tsx
"use client";

import { useState, useCallback } from "react";
import { useRouter } from "next/navigation";
import { articleApi } from "@/lib/api-client";
import type { CreateArticleInput } from "@repo/core";

type FormState = {
  readonly title: string;
  readonly content: string;
  readonly tags: string;
  readonly isSubmitting: boolean;
  readonly error: string | null;
};

const INITIAL_STATE: FormState = {
  title: "",
  content: "",
  tags: "",
  isSubmitting: false,
  error: null,
};

// 상태 변환 함수 — React에서도 불변 데이터 + 순수 함수 패턴
const updateField = <K extends keyof FormState>(
  state: FormState,
  field: K,
  value: FormState[K],
): FormState => ({ ...state, [field]: value, error: null });

const parseTags = (input: string): readonly string[] =>
  input
    .split(",")
    .map((tag) => tag.trim())
    .filter((tag) => tag.length > 0);

const toCreateInput = (state: FormState): CreateArticleInput => ({
  title: state.title,
  content: state.content,
  tags: parseTags(state.tags),
});

export default function NewArticlePage() {
  const router = useRouter();
  const [state, setState] = useState<FormState>(INITIAL_STATE);

  const handleSubmit = useCallback(
    async (e: React.FormEvent) => {
      e.preventDefault();
      setState((s) => ({ ...s, isSubmitting: true, error: null }));

      const token = ""; // 실제로는 인증 토큰
      const result = await articleApi.create(toCreateInput(state), token);

      if (result.ok) {
        router.push("/");
      } else {
        setState((s) => ({
          ...s,
          isSubmitting: false,
          error: result.error.message ?? "작성에 실패했습니다",
        }));
      }
    },
    [state, router],
  );

  return (
    <main className="mx-auto max-w-3xl px-4 py-8">
      <h1 className="mb-8 text-3xl font-bold">새 기사 작성</h1>

      <form onSubmit={handleSubmit} className="space-y-6">
        <div>
          <label htmlFor="title" className="block text-sm font-medium">
            제목
          </label>
          <input
            id="title"
            type="text"
            value={state.title}
            onChange={(e) =>
              setState((s) => updateField(s, "title", e.target.value))
            }
            className="mt-1 w-full rounded border px-3 py-2"
            required
          />
        </div>

        <div>
          <label htmlFor="content" className="block text-sm font-medium">
            본문
          </label>
          <textarea
            id="content"
            value={state.content}
            onChange={(e) =>
              setState((s) => updateField(s, "content", e.target.value))
            }
            rows={15}
            className="mt-1 w-full rounded border px-3 py-2"
            required
          />
        </div>

        <div>
          <label htmlFor="tags" className="block text-sm font-medium">
            태그 (쉼표로 구분)
          </label>
          <input
            id="tags"
            type="text"
            value={state.tags}
            onChange={(e) =>
              setState((s) => updateField(s, "tags", e.target.value))
            }
            placeholder="예: typescript, 아키텍처, 클린코드"
            className="mt-1 w-full rounded border px-3 py-2"
          />
        </div>

        {state.error && (
          <div className="rounded bg-red-50 p-3 text-red-700">
            {state.error}
          </div>
        )}

        <button
          type="submit"
          disabled={state.isSubmitting}
          className="rounded bg-blue-600 px-6 py-2 text-white hover:bg-blue-700 disabled:opacity-50"
        >
          {state.isSubmitting ? "저장 중..." : "저장"}
        </button>
      </form>
    </main>
  );
}
```

## 핵심 설계 원칙 요약

**DOP 원칙 적용:**
- 코드(함수)와 데이터(타입/레코드)를 분리했다
- 모든 데이터는 일반 레코드(plain object)다 — class 없음
- readonly로 불변성을 강제했다
- DB 스키마(drizzle)와 도메인 모델을 매퍼로 분리했다

**FP 원칙 적용:**
- `neverthrow`의 Result/ResultAsync로 예외 대신 값으로 에러를 표현한다
- `remeda`의 pipe/map/unique/clamp로 컬렉션 변환을 선형으로 유지한다
- `ts-pattern`의 match().exhaustive()로 판별 합집합 분기를 안전하게 소비한다
- 판별 합집합(discriminated union)으로 불가능한 상태를 표현 불가능하게 만들었다
- 커링된 팩토리 함수로 의존성을 주입한다 (Reader 패턴)
- 상태 전이는 순수 함수다 — 입력을 받아 새 데이터를 반환한다

**Clean Architecture 적용:**
- 의존성은 안쪽(core)을 향한다 — core는 외부를 모른다
- 포트(함수 타입 레코드)와 어댑터(팩토리 함수가 반환하는 객체)로 경계를 나눈다
- 컴포지션 루트(`deps.ts`)에서 모든 것을 조립한다
